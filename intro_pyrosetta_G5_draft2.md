### 3.2 Scoring

A score function is an energy function that assigns a numerical score to the pose. It is a weighted combination of physical and statistical terms where each term conrresponds to specific interactions eg van der waals or hydrogen bonding intercations.To score a protein we need to define a scoring function using **get_fa_scorefxn()**. This tells Rosetta to load the default full-atom energy terms.
To get the energy of the mpro protein we can use this function.


```python
scorefxn = get_fa_scorefxn()
print(scorefxn(pose))
```

The score of this protein is -136.2613147713958

#### 3.2.1 Importance of relaxing your pdb
PDB structures are almost never in the energy minimum of the Rosetta energy function.
Relaxing your pose lets Rosetta adjust the structure so the energy you compute is meaningful and comparable. We will use the **FastRelax** method in pyrosetta to relax our pose. 

While we want to put the sturcture in its lowest energy state, we want to keep the backbone information from the crystal structure as much as possible (lowest RMSD). 
Therefore, we apply constrain_relax_to_start_coords(True) to FastRelax().

**NOTE**: FastRelax() does take up a huge amount of resource, so be concious of the size of the structure you are working with and your computational resources.
first we want to clone our pose so we dont make changes to the old pose



```python
new_pose = pose.clone()
relax = FastRelax()
relax.set_scorefxn(scorefxn)
relax.constrain_relax_to_start_coords(True)
relax.apply(new_pose)
print(scorefxn(new_pose))
```

The energy output for the new pose is -954.3207875360717

Now we can save this relaxed pose to a pdb file using the **dump_pdb()** method



```python
new_pose.dump_pdb("relaxed_starting_structure.pdb")
```

You should now have a file names **relaxed_starting_structure.pdb** in your current working directory

You can load both the original unrelaxed (6M03.pdb) and the relaxed structure (relaxed_starting_structure.pdb) in pymol to visualize the difference. it should look different

def mutate_and_score(pose, position, new_amino_acid):
    """
    Mutate a single residue using the toolbox function.
    
    Args:
        pose: PyRosetta Pose object
        position: Residue number to mutate (1-indexed)
        new_amino_acid: Single letter code (e.g., 'A' for Alanine)
    
    Returns:
        Dictionary with mutation info and energies
    """
    original_aa = pose.residue(position).name1()
    original_energy = scorefxn(pose)
    
    # Clone the pose
    mutant_pose = pose.clone()
    
    # Use toolbox function - this mutates AND repacks neighbors, 
    #the pack_radius is the radius of the sphere around the mutated residue 
    #that will be repacked (5 Angstroms is a this function's default)
    mutate_residue(mutant_pose, position, new_amino_acid, pack_radius=5.0)
    new_energy = scorefxn(mutant_pose)
    
    return {
        'position': position,
        'original_aa': original_aa,
        'new_aa': new_amino_acid,
        'original_energy': original_energy,
        'mutant_energy': new_energy,
        'delta_energy': new_energy - original_energy,
        'mutant_pose': mutant_pose
    }
def scan_one_position_many_aas(pose, position, amino_acids, output_csv="scan_one_pos.csv"):
    """
    Mutate ONE position to MANY different amino acids.
    
    Args:
        pose: PyRosetta Pose object
        position: Single residue position (e.g., 50)
        amino_acids: List of amino acids to try (e.g., ['A', 'G', 'V', 'L'])
        output_csv: Output filename
    """
    results = []
    
    with open(output_csv, 'w', newline='') as f:
        writer = csv.DictWriter(f, fieldnames=['position', 'original_aa', 'new_aa', 
                                                'original_energy', 'mutant_energy', 'delta_energy'])
        writer.writeheader()
        
        for aa in amino_acids:
            print(f"Mutating position {position} to {aa}...")
            result = mutate_and_score(pose, position, aa)
            
            # Write row (exclude pose object)
            row = {k: v for k, v in result.items() if k != 'mutant_pose'}
            writer.writerow(row)
            results.append(result)
            
            mutant_pose = result['mutant_pose']
            mutant_pose.dump_pdb(f"position{position}_{result['original_aa']}{position}{aa}.pdb")
    
    print(f"Done! Results saved to {output_csv}")
    return results
    
def scan_many_positions_one_aa(pose, positions, amino_acid, output_csv="scan_many_pos.csv"):
    """

            results.append(result)

def apply_specific_mutations(pose, mutations, output_csv):

def run_simple_repack(pose):
    """
    run_enristricted_repack_with_no_design
    """
    task_factory = TaskFactory()


    task_factory.push_back(InitializeFromCommandline())
    task_factory.push_back(RestrictToRepacking())  # Only repack, no design

    packer = PackRotamersMover()
    packer.score_function(scorefxn)
    packer.task_factory(task_factory)
    packer.apply(pose)

def compute_binding_deltaG(pose, repeats=3, jump_id=1):

    #Computes bound energy assumes input pose is docked and or in the known correct dimer position
    bound_energies = []
    for i in range(repeats):
        bound_pose = pose.clone()
        run_simple_repack(bound_pose)
        energy_bound = scorefxn(bound_pose)
        bound_energies.append(energy_bound)

    # We move the partners apart once to create the "base" unbound pose
    unbound_base = pose.clone()
    trans_mover = RigidBodyTransMover(unbound_base, jump_id)
    trans_mover.step_size(500.0) #Push them 500Å away
    trans_mover.apply(unbound_base)

    #Computes unbound energies
    unbound_energies = []
    for i in range(repeats):
        unbound_pose = unbound_base.clone()
        run_simple_repack(unbound_pose)
        energy_unbound = scorefxn(unbound_pose)
        unbound_energies.append(energy_unbound)

    # --- Calculation ---
    avg_bound = sum(bound_energies) / len(bound_energies)
    avg_unbound = sum(unbound_energies) / len(unbound_energies)

    ddg = avg_bound - avg_unbound
    if ddg < 0:
        return True , ddg
    else:
        return False , ddg

def run_min_remove_clashes(pose):
    """
    Minimizes the pose using 'ref2015_soft' to resolve steric clashes 
    without pushing the chains apart.
    """
    print("\n[PRE-CLEAN] Running Soft-Repulsion Minimization...")
    
    # 1. Use the Soft Score Function (allows slight overlaps)
    soft_scorefxn = create_score_function("ref2015_soft")
    
    # 2. Setup MoveMap
    # BB/CHI = True: Allow atoms to wiggle to fit
    # JUMP = False: CRITICAL! Locks the chains together so they can't fly apart
    mm = MoveMap()
    mm.set_bb(True)
    mm.set_chi(True)
    mm.set_jump(False) 
    
    # 3. Configure Minimizer
    min_mover = MinMover()
    min_mover.movemap(mm)
    min_mover.score_function(soft_scorefxn)
    min_mover.tolerance(0.01)
    
    # 4. Run
    min_mover.apply(pose)

    min_mover.score_function(scorefxn)
    min_mover.apply(pose)
